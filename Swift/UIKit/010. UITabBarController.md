## UITabBarController



Программная настройка отображения вкладок, убираем прозрачность:

```swift
let tabBarAppearance = UITabBarAppearance()
tabBarAppearance.configureWithOpaqueBackground()
tabBar.standardAppearance = tabBarAppearance
tabBar.scrollEdgeAppearance = tabBarAppearance
```



Задать заголовок вкладки программно:

```swift
tabBar.items?.last?.title = user.person.fullname
```



Для доступа к вложенным View Controller нужно обратиться к массиву `viewControllers`:

```swift
import UIKit

final class TabBarController: UITabBarController {

    var user: User!
    
    override func viewDidLoad() {
        super.viewDidLoad()

        let tabBarAppearance = UITabBarAppearance()
        tabBarAppearance.configureWithOpaqueBackground()
        tabBar.standardAppearance = tabBarAppearance
        tabBar.scrollEdgeAppearance = tabBarAppearance
        
        tabBar.items?.last?.title = user.person.fullname
        
        viewControllers?.forEach { viewController in
            if let welcomVC = viewController as? WelcomeViewController {
                welcomVC.user = user
            } else if let navigationVC = viewController as? UINavigationController {
                
                if let personVC = navigationVC.topViewController as? PersonViewController {
                    personVC.user = user
                }
            }
        }
        
        
    }
    
}
```
