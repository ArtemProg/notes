#### –û–±—Ä–∞–±–æ—Ç–∫–∞ –æ—à–∏–±–æ–∫

```swift

    var toner: Bool

    case noPaper

    guard printer.paper else {
        throw PrintingErrors.noPaper
    }
    
    guard printer.toner else {
        throw PrintingErrors.noToner
    }
    
    print("–î–æ–∫—É–º–µ–Ω—Ç —É—Å–ø–µ—à–Ω–æ –Ω–∞–ø–µ—á–∞—Ç–∞–Ω!")
```
